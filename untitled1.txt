<script>
    // أ. المعطيات الأساسية من نموذج جوجل (رابط الإرسال ومعرفات الإدخال - Entry IDs)
    // هذا هو الرابط الذي يستقبله نموذج جوجل عند إرسال البيانات (ينتهي بـ /formResponse)
    const googleFormUrl = "https://docs.google.com/forms/d/e/1FAIpQLSfOMCIhwk6-yrqu-yx-sxwHKjFSLDPsmnyEcWAod6Ql6zxsiw/formResponse";

    // معرفات الإدخال الفريدة لكل حقل في نموذج جوجل، بناءً على تخطيطنا الذكي
    const entryIds = {
        // البيانات الأساسية (5 حقول)
        name: "entry.1112362374",      // الاسم رباعي باللغة الإنجليزية
        dob: "entry.430574773",       // تاريخ الميلاد
        gender: "entry.444597116",    // النوع (ذكر/أنثى)
        city: "entry.714961820",      // المحافظة
        phone: "entry.380930077",     // رقم الهاتف (الواتساب)
        
        // معرف الفئة العمرية في جوجل (يُستخدم للتحقق ولكن لا يُرسل كقيمة تحدي)
        ageCategory: "entry.1888533519",

        // معرفات التحديات والباقات بناءً على الفئة العمرية (4 حقول)
        // قمنا بتخطيط حقول "الجري" في جوجل لتقابل "جونيور" وحقول "الدراجات" لتقابل "سينيور" في واجهتك
        junior: {
            challenge: "entry.1995806128", // معرف تحدي الجري في جوجل (يقابل تحدي الجونيور في واجهتك)
            package: "entry.185047638"     // معرف باقة الجري في جوجل (يقابل باقة الجونيور في واجهتك)
        },
        senior: {
            challenge: "entry.762355901",   // معرف تحدي الدراجات في جوجل (يقابل تحدي السينيور في واجهتك)
            package: "entry.1842057656"     // معرف باقة الدراجات في جوجل (يقابل باقة السينيور في واجهتك)
        }
    };

    // ب. البيانات الديناميكية للتحديات والباقات (للعرض في الواجهة)
    const data = {
        senior: {
            challenges: [
                { val: 'cycling_300', text: 'دراجات (300 كم)' },
                { val: 'running_120', text: 'جري (120 كم)' },
                { val: 'walking_100', text: 'مشي (100 كم)' },
                { val: 'skating_200', text: 'سبيد سكيتر (200 كم)' }
            ],
            packages: [
                { val: 's_100', text: 'ميدالية (100 ج.م)' },
                { val: 's_300', text: 'ميدالية + تيشرت (300 ج.م)' },
                { val: 's_450', text: 'ميدالية + حامل ميداليات (450 ج.م)' },
                { val: 's_600', text: 'باقة 4 قطع (600 ج.م)' },
                { val: 's_750', text: 'الباقة الشاملة (750 ج.م)' }
            ]
        },
        junior: {
            challenges: [
                { val: 'junior_run_50k', text: 'تحدي الجونيور (50 كم جري)' }
            ],
            packages: [
                { val: 'j_100', text: 'ميدالية (100 ج.م)' },
                { val: 'j_250', text: 'ميدالية + تيشرت كَت (250 ج.م)' }
            ]
        }
    };

    // ج. الدوال المساعدة للواجهة (الديناميكية)

    // الدالة دي بتشتغل بمجرد ما المتسابق يختار الفئة العمرية
    function updateDynamicInputs() {
        const category = document.getElementById('category').value;
        const challengeSelect = document.getElementById('challengeType');
        const packageSelect = document.getElementById('packageType');

        // تفريغ الخانات تماماً
        challengeSelect.innerHTML = '';
        packageSelect.innerHTML = '';

        if (category) {
            // فتح الخانات للمتسابق
            challengeSelect.disabled = false;
            packageSelect.disabled = false;

            // إضافة الخيار الأول (Placeholder)
            challengeSelect.add(new Option('-- اختر نوع التحدي --', ''));
            packageSelect.add(new Option('-- اختر باقة الاشتراك --', ''));

            // تعبئة البيانات باستخدام الطريقة المدعومة في كل المتصفحات
            data[category].challenges.forEach(item => {
                challengeSelect.add(new Option(item.text, item.val));
            });

            data[category].packages.forEach(item => {
                packageSelect.add(new Option(item.text, item.val));
            });

            // لو اختار جونيور، نختار الجري أوتوماتيك عشان هو اختيار واحد
            if (category === 'junior') {
                challengeSelect.value = 'junior_run_50k';
            }
        } else {
            // قفل الخانات لو رجع اختار الفاضي
            challengeSelect.disabled = true;
            packageSelect.disabled = true;
            challengeSelect.add(new Option('-- مقفول: اختر الفئة العمرية أولاً --', ''));
            packageSelect.add(new Option('-- مقفول: اختر الفئة العمرية أولاً --', ''));
        }
    }

    // د. معالجة إرسال البيانات (Form Submission & Fetch)

    const myForm = document.getElementById("regForm");
    const nameInput = document.getElementById("name");
    const dobInput = document.getElementById("dob");
    const genderInput = document.getElementById("gender");
    const govInput = document.getElementById("governorate");
    const phoneInput = document.getElementById("phone");
    const categorySelect = document.getElementById("category");
    const challengeSelect = document.getElementById("challengeType");
    const packageSelect = document.getElementById("packageType");
    const submitButton = document.getElementById("submitBtn");

    myForm.addEventListener("submit", function(e) {
      e.preventDefault(); // منع السلوك الافتراضي لإعادة تحميل الصفحة

      // تغيير حالة الزر لإظهار جاري الإرسال ومنعه من الضغط المتكرر
      submitButton.textContent = "جاري الإرسال...";
      submitButton.disabled = true;

      // 1. التحقق من صحة الاسم باللغة الإنجليزية فقط (Regex)
      const englishNameRegex = /^[a-zA-Z\s]+$/;
      if (!englishNameRegex.test(nameInput.value)) {
        alert("عذراً، يجب كتابة الاسم باللغة الإنجليزية فقط لضمان طباعته بشكل صحيح على الشهادات والميداليات.");
        resetButton(); // إعادة الزر لحالته الأصلية
        return;
      }

      // الحصول على القيم المحددة
      const selectedAgeCat = categorySelect.value;
      const selectedChallenge = challengeSelect.value;
      const selectedPackage = packageSelect.value;

      // التحقق من أن جميع الاختيارات قد تمت
      if (!selectedAgeCat || !selectedChallenge || !selectedPackage) {
          alert("يرجى التأكد من اختيار الفئة العمرية ونوع التحدي والباقة أولاً.");
          resetButton();
          return;
      }

      // 2. تجهيز البيانات للإرسال باستخدام FormData
      const formData = new FormData();

      // أ. إضافة البيانات الأساسية (5 حقول)
      formData.append(entryIds.name, nameInput.value);
      formData.append(entryIds.dob, dobInput.value);
      formData.append(entryIds.gender, genderInput.value);
      formData.append(entryIds.city, govInput.value);
      formData.append(entryIds.phone, phoneInput.value);

      // ب. المنطق البرمجي الذكي لإرسال التحدي والباقة للفئة الصحيحة (CORS logic)
      // نقوم بفحص الفئة العمرية التي اختارها المتسابق (جونيور/سينيور)
      if (selectedAgeCat === "junior") {
          // إذا كان جونيور: نرسل التحدي والباقة إلى معرفات الجونيور في جوجل، ونترك معرفات السينيور فارغة
          formData.append(entryIds.junior.challenge, selectedChallenge); // يرسل مثلاً "junior_run_50k" إلى معرف الجري في جوجل
          formData.append(entryIds.junior.package, selectedPackage);
          formData.append(entryIds.senior.challenge, ""); // يترك حقل تحدي الدراجات في جوجل فارغاً
          formData.append(entryIds.senior.package, "");
      } else if (selectedAgeCat === "senior") {
          // إذا كان سينيور: نرسل التحدي والباقة إلى معرفات السينيور في جوجل، ونترك معرفات الجونيور فارغة
          formData.append(entryIds.junior.challenge, ""); // يترك حقل تحدي الجري في جوجل فارغاً
          formData.append(entryIds.junior.package, "");
          formData.append(entryIds.senior.challenge, selectedChallenge); // يرسل مثلاً "cycling_300" إلى معرف الدراجات في جوجل
          formData.append(entryIds.senior.package, selectedPackage);
      }

      // 3. إرسال البيانات باستخدام fetch
      // ملاحظة: نستخدم mode: 'no-cors' لأن نماذج جوجل لا تدعم CORS. 
      // هذا يعني أننا لن نستطيع قراءة استجابة جوجل، لذا سنعتبر الإرسال نجح بمجرد إتمامه.
      fetch(googleFormUrl, {
        method: "POST",
        mode: "no-cors",
        body: formData
      })
      .then(() => {
        // بما أننا نستخدم no-cors، سنفترض النجاح هنا بعد الإرسال
        console.log("تم إرسال البيانات بنجاح إلى نموذج جوجل!");
        alert("شكراً لتسجيلك! تم استلام بياناتك بنجاح.");
        myForm.reset(); // تفريغ الفورم
        // إعادة تعيين القوائم الديناميكية
        updateDynamicInputs(); 
        resetButton(); // إعادة الزر لحالته الأصلية
      })
      .catch(error => {
        // هذا الجزء نادراً ما يتم الوصول إليه مع no-cors إلا في حال فشل الاتصال بالشبكة
        console.error("حدث خطأ في الشبكة أثناء الإرسال:", error);
        alert("عذراً، حدث خطأ في الشبكة أثناء إرسال البيانات. يرجى المحاولة لاحقاً.");
        resetButton(); // إعادة الزر لحالته الأصلية
      });
    });

    // دالة مساعدة لإعادة الزر لحالته الأصلية
    function resetButton() {
        submitButton.textContent = "تأكيد التسجيل";
        submitButton.disabled = false;
    }
</script>
